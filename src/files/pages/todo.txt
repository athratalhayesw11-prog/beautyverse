  /* keep ref synced */

    dotsCountRef.current = dotsCount;
  }, [dotsCount]);

  /* responsive dots count */
  useEffect(() => {
    const updateDots = () => {
      if (window.innerWidth <= 600) {
        setDotsCount(stylists.length); // 1 dot per card
      } else {
        setDotsCount(3); // desktop sections
      }
    };

    updateDots();
    window.addEventListener("resize", updateDots);
    return () => window.removeEventListener("resize", updateDots);
  }, []);

  /* üî• SCROLL HANDLER */
  const handleScroll = () => {
    if (!stylistsRef.current) return;

    const carousel = stylistsRef.current;
    const scrollLeft = carousel.scrollLeft;

    /* üì± MOBILE ‚Üí CARD BASED */
    if (window.innerWidth <= 600) {
      const cardFullWidth = CARD_WIDTH + CARD_GAP;
      const activeIndex = Math.round(scrollLeft / cardFullWidth);

      setActiveDots([Math.min(activeIndex, stylists.length - 1)]);
      return;
    }

    /* üñ• DESKTOP ‚Üí PERCENTAGE BASED */
    const maxScroll = carousel.scrollWidth - carousel.clientWidth;
    if (maxScroll <= 0) return;

    const progress = scrollLeft / maxScroll;
    const step = 1 / dotsCountRef.current;

    const activeIndex = Math.min(
      dotsCountRef.current - 1,
      Math.floor(progress / step)
    );

    setActiveDots([activeIndex]);
  };

  /* attach scroll listener */
  useEffect(() => {
    const carousel = stylistsRef.current;
    if (!carousel) return;

    carousel.addEventListener("scroll", handleScroll);
    return () => carousel.removeEventListener("scroll", handleScroll);
  }, []);

  /* arrows */
  const scrollByCard = (dir) => {
    if (!stylistsRef.current) return;

    const carousel = stylistsRef.current;
    const amount = (CARD_WIDTH + CARD_GAP) * dir;

    carousel.scrollTo({
      left: carousel.scrollLeft + amount,
      behavior: "smooth"
    });
  };

  const handleBookStylist = (stylist) => {
    alert(`Booking appointment with ${stylist.name} - ${stylist.specialty}`);
  };

        </h2>
        <p className="services-subtitle">
          Experience beauty transformation with our certified professionals
        </p>
      </div>

      <div className="carousel-container">
        <button className="nav-button prev-button" onClick={() => scrollByCard(-1)}>
          <img src={left} alt="" />
        </button>

        <div className="stylists-carousel" ref={stylistsRef}>
          {stylists.map((stylist) => (
            <div className="stylist-card" key={stylist.id}>
              <img src={stylist.image} alt={stylist.name} className="profile-image" />
              <h3>{stylist.name}</h3>
              <p>{stylist.description}</p>
              <button onClick={() => handleBookStylist(stylist)}>Book Stylist</button>
            </div>
          ))}
        </div>

        <button className="nav-button next-button" onClick={() => scrollByCard(1)}>
          <img src={right} alt="" />
        </button>
      </div>

        {Array.from({ length: dotsCount }).map((_, index) => (
          <button
            key={index}
            className={`dot ${activeDots[0] === index ? "active" : ""}`}
          />
        ))}
      </div>
    </section>
